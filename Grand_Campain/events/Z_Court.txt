# The Emerald Buddha by Daniel Tolman
country_event = {
	id = z_court.1
	title = z_court.1.t
	desc = z_court.1.d
	picture = BYZANTINE_EMPEROR_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		religion_group = pagan
		FROM = {
			OR = {
				has_province_modifier = silla_court
			}
			NOT = {
				owned_by = ROOT
			}
			NOT = {
				owner = {
					OR = {
						is_in_war = {
							defenders = PREV
							defenders = ROOT
						}
						is_in_war = {
							attackers = PREV
							attackers = ROOT
						}
					}
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			FROM = {
				save_global_event_target_as = silla_court_province
				remove_province_modifier = silla_court
				add_permanent_province_modifier = {
					name = silla_court_shadow
					duration = 3650
				}
			}
		}
	}
	
	option = {
		name = z_court.1.a
		add_prestige = 10
		hidden_effect = {
			FROM = {
				owner = {
					country_event = {
						id = z_court.2
					}
				}
			}
		}
		capital_scope = {
			add_permanent_province_modifier = {
				name = silla_court
				duration = -1
			}
		}
	}
}

# Emerald Buddha lost
country_event = {
	id = z_court.2
	title = z_court.2.t
	desc = z_court.2.d
	picture = COUNTRY_COLLAPSE_eventPicture
	
	is_triggered_only = yes
	
	option = {
		name = z_court.2.a
		if = {
			limit = {
				religion_group = pagan
			}
			add_prestige = -20
			add_legitimacy = -10
		}
		else = {
			add_prestige = -5
		}
	}
	
	after = {
		clear_global_event_target = silla_court_province
	}
}
