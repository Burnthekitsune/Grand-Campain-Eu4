cultural_issues = {
	slot = 3
	generic = no
	ai = yes
	potential = {
		tag = Z0H
	}
	has_country_shield = yes

	a_cultural_question = {
		icon = mission_dominate_home_trade_node
		position = 1
		required_missions = {  }
		trigger = {
			OR = {
				marriage_with = SPL
				marriage_with = NAP
				is_year = 1480
			}
		}
		effect = {
			if = {
				limit = {
					marriage_with = SPL
					NOT = {
						marriage_with = NAP
					}
				}
				set_county_flag = alsamali_way
				vassalize = SPL
				complete_mission = bring_in_spoleto
				swap_non_generic_missions = yes
			}
			
			else_if = {
				limit = {
					marriage_with = NAP
					NOT = {
						marriage_with = SPL
					}
				}
				set_county_flag = italian_way
				swap_non_generic_missions = yes
				vassalize = NAP
				change_culture = umbrian
				set_ruler_culture = umbrian
				set_heir_culture = umbrian
				set_consort_culture = umbrian
				119 = {
					add_permanent_claim = ROOT
				}
				120 = {
					add_permanent_claim = ROOT
				}
				4731 = {
					add_permanent_claim = ROOT
				}
				complete_mission = bring_in_naples
				}
			
			else = {
				set_county_flag = balanced_way
				swap_non_generic_missions = yes
				if = {
					limit = {
						4731 = {
							owned_by = SPL
						}
					}
					4731 = {
						cede_province = ROOT
						add_core = ROOT
					}
				}
				if = {
					limit = {
						4733 = {
							owned_by = NAP
						}
					}
					4733 = {
						cede_province = ROOT
						add_core = ROOT
					}
				}
				naples_area = {
					add_permanent_claim = ROOT
				}
				calabria_area = {
					add_permanent_claim = ROOT
				}
				apulia_area = {
					add_permanent_claim = ROOT
				}
				central_italy_area = {
					add_permanent_claim = ROOT
				}
				lazio_area = {
					add_permanent_claim = ROOT
				}
				complete_mission = sway_the_borders
			}
		}
	}
}
